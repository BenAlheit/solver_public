139 template<unsigned int dim>
140 void CubeWithSpherePBC<dim>::read_and_refine_mesh() {
141  GridIn<dim> grid_in;
142  grid_in.attach_triangulation(triangulation);
143  ifstream input_file("../src/examples/cube-with-particle.ucd");
144  grid_in.read_ucd(input_file);
145  cout << "Read mesh" << endl;
146  triangulation.reset_all_manifolds();
147  triangulation.set_all_manifold_ids(0);
148  Point<dim> centre(0.0, 0.0, 0.5);
149  for (const auto &cell: triangulation.cell_iterators()) {
150  if (cell->material_id() == 1) {
151  for (const auto &face: cell->face_iterators()) {
152  bool face_at_sphere_boundary = true;
153  for (const auto v: face->vertex_indices()) {
154  if (fabs((face->vertex(v) - centre).norm() - 0.25) > 1e-3) {
155  face_at_sphere_boundary = false;
156  break;
157  }
158  }
159  if (face_at_sphere_boundary)
160  face->set_all_manifold_ids(1);
161  }
162  }
163  }
164 
165  Point<dim> cell_centre;
166 
167  for (const auto &cell: triangulation.cell_iterators()) {
168  cell_centre = cell->center();
169  if ((cell_centre - centre).norm() > 0.25) {
170  cell->set_material_id(0);
171  } else {
172  cell->set_material_id(1);
173  }
174  }
175 
176  for (auto &face: triangulation.active_face_iterators()) {
177  if (face->at_boundary()) {
178  for (const auto &i: range) {
179  if (almost_equals(face->center()[i], p1[i])) {
180  face->set_boundary_id(1 + i);
181  break;
182  } else if (almost_equals(face->center()[i], p2[i])) {
183  face->set_boundary_id(1 + dim + i);
184  break;
185  }
186  }
187  }
188  }
189 
190  triangulation.set_manifold(1, SphericalManifold<dim>(centre));
191  TransfiniteInterpolationManifold<dim> transfinite_manifold;
192  transfinite_manifold.initialize(triangulation);
193  triangulation.set_manifold(0, transfinite_manifold);
194 
195  if (n_refinements > 0) triangulation.refine_global(n_refinements);
196 
197  cout << "Refined mesh" << endl;
198 
199 
200  ofstream out("cube-with-sphere-" + to_string(n_refinements) + ".vtk");
201  GridOut grid_out;
202  grid_out.write_vtk(triangulation, out);
203 
204  cout << "Written mesh" << endl;
205 
206 }
