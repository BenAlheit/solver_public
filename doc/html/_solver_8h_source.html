169  void Solver<dim>::assemble_system(bool init_step, bool init_solve) {
170  system_matrix = 0.0;
171  system_rhs = 0.0;
172 
173  const Vector<double> localized_solution(u);
174 
175  FullMatrix<double> cell_matrix(problem->get_mesh()->get_dofs_per_cell(),
176  problem->get_mesh()->get_dofs_per_cell());
177  Vector<double> cell_rhs(problem->get_mesh()->get_dofs_per_cell());
178 
179  vector<types::global_dof_index> local_dof_indices(problem->get_mesh()->get_dofs_per_cell());
180  StateBase<dim> *qp_state;
181  Material<dim> *material;
182  array<unsigned int, dim> i_node_local_dofs, j_node_local_dofs;
183 
184  vector<Tensor<1, dim>> spatial_grad_phis(problem->get_mesh()->get_n_shape_fns());
185  vector<Tensor<1, dim>> ref_grad_phis(problem->get_mesh()->get_n_shape_fns());
186 
187  Tensor<1, dim> grad_phi_i, grad_phi_j, r_i;
188  Tensor<2, dim> Grad_u, F, F_inv, tau, k_ij;
189  SymmetricTensor<2, dim> sym_tau;
190  SymmetricTensor<4, dim> sym_tangent;
191  Tensor<3, dim> grad_phi_i_tangent;
192  Tensor<1, dim> grad_phi_i_tau;
193  SymmetricTensor<2, dim> I = Physics::Elasticity::StandardTensors<dim>::I;
194  double JxW;
195 
196  Vector<double> local_u;
197  local_u.reinit(problem->get_mesh()->get_dofs_per_cell());
198 
199  for (const auto &cell: problem->get_mesh()->get_dof_handler().active_cell_iterators())
200  if (cell->subdomain_id() == problem->get_mesh()->get_this_mpi_processes()) {
201  cell_matrix = 0;
202  cell_rhs = 0;
203  cell->get_dof_indices(local_dof_indices);
204  for (const auto &i_dof: problem->get_mesh()->get_dof_range())
205  local_u[i_dof] = localized_solution[local_dof_indices[i_dof]];
206 
207 
208  for (const auto &qp: problem->get_mesh()->get_qp_range()) {
209  JxW = problem->get_mesh()->get_JxW(cell->level(), cell->index(), qp);
210  Grad_u = problem->get_mesh()->grad_field(cell->level(), cell->index(), qp, local_u);
211  F = I + Grad_u;
212  F_inv = invert(F);
213  qp_state = problem->get_mesh()->get_state(cell->level(), cell->index(), qp);
214 
215  qp_state->F_n1 = F;
216 
217  if (not init_step or init_solve) {
218  material = problem->get_mesh()->get_material(cell->material_id());
219  material->set_state(qp_state);
220  material->update_stress_and_tangent(problem->get_time()->get_current_delta_t());
221  }
222 
223  sym_tau = symmetrize(qp_state->tau_n1);
224 
225  ref_grad_phis = problem->get_mesh()->get_grad_shape_fns(cell->level(), cell->index(), qp);
226  for (const auto &i_node: problem->get_mesh()->get_local_nodes())
227  spatial_grad_phis.at(i_node) = ref_grad_phis.at(i_node) * F_inv;
228 
229  for (const auto &i_node: problem->get_mesh()->get_local_nodes()) {
230  i_node_local_dofs = problem->get_mesh()->get_local_node_to_dofs(i_node);
231  grad_phi_i = spatial_grad_phis.at(i_node);
232 
233  grad_phi_i_tau = sym_tau * grad_phi_i;
234  grad_phi_i_tangent = grad_phi_i * qp_state->c_n1;
235 
236  r_i = -grad_phi_i_tau * JxW;
237  for (const auto &i: range)
238  cell_rhs[i_node_local_dofs.at(i)] += r_i[i];
239 
240  for (const auto &j_node: problem->get_mesh()->get_local_nodes()) {
241  j_node_local_dofs = problem->get_mesh()->get_local_node_to_dofs(j_node);
242  grad_phi_j = spatial_grad_phis.at(j_node);
243  k_ij = 0;
244  k_ij = (grad_phi_i_tangent * grad_phi_j + I * (grad_phi_i_tau * grad_phi_j)) * JxW;
245 
246  for (const auto &i: range)
247  for (const auto &j: range)
248  cell_matrix[i_node_local_dofs.at(i)][j_node_local_dofs.at(j)] += k_ij[i][j];
249  }
250  }
251  }
252 
253  du_constraints.distribute_local_to_global(cell_matrix,
254  cell_rhs,
255  local_dof_indices,
256  system_matrix,
257  system_rhs);
258  }
259 
260  system_matrix.compress(VectorOperation::add);
261  system_rhs.compress(VectorOperation::add);
262
