262 Vector<double> RungeKutta::integrate(const Vector<double> &y_t_start,
263  const function<Vector<double>(const double &, const Vector<double> &)> &dy) {
264  Vector<double> y_n1(y_t_start.size()), y_n(y_t_start.size()), dy_n(y_t_start.size());
265  y_n = y_t_start;
266  vector<unsigned int> steps(this->n_steps);
267  iota(steps.begin(), steps.end(), 0);
268  double step_dt = (this->t_end - this->t_start) / this->n_steps;
269  this->t_c_start = this->t_start;
270  for (const auto &n: steps) {
271  this->t_c_end = this->t_c_start + step_dt;
272  y_n1 = integrate_step(y_n, dy);
273  y_n = y_n1;
274  this->t_c_start += step_dt;
275  }
276 
277  return y_n1;
278 }

280 Vector<double> RungeKutta::integrate_step(const Vector<double> &y_t_start,
281  const function<Vector<double>(const double &, const Vector<double> &)> &dy) {
282  unsigned int n_dofs = y_t_start.size();
283  Vector<double> out(n_dofs);
284  Vector<double> error_approx(n_dofs);
285  Vector<double> working(n_dofs);
286  Vector<double> dy_i(n_dofs);
287  unsigned int s = tableau->c.size();
288  vector<unsigned int> range(s);
289 // vector<unsigned int> range_b(tableau->b.at(0).size());
290  vector<Vector<double>> ks(s, Vector<double>(n_dofs));
291  iota(range.begin(), range.end(), 0);
292  double dt = this->t_c_end - this->t_c_start;
293 
294  for (const auto &i: range) {
295  working = y_t_start;
296  for (unsigned int j = 0; j < i; j++) {
297  working.add(dt * tableau->a.at(i).at(j), ks.at(j));
298  }
299  dy_i = dy(t_c_start + tableau->c.at(i) * dt, working);
300  ks.at(i) = dy_i;
301  }
302  out = y_t_start;
303  for (const auto &i: range)
304  out.add(dt * tableau->b.at(0).at(i), ks.at(i));
305 
306  error_approx = 0;
307  for (const auto &i: range)
308  error_approx.add(dt * (tableau->b.at(0).at(i) - tableau->b.at(1).at(i)), ks.at(i));
309 
310  if (error_approx.l2_norm() / out.l2_norm() > 0.0001) {
311  cout << "Large error. L2 norm ratio: " << error_approx.l2_norm() / out.l2_norm() << endl;
312  cout << "Error vector: " << endl;
313  error_approx.print(cout);
314  cout << "Solution vector: " << endl;
315  out.print(cout);
316  cout << "Flow error %: " << 100 * error_approx[9] / out[9] << endl;
317  }
318 
319  return out;
320 }
